218 bool RF_reciever::convertBitToMessage(byte currentBit)
219 {
220  //increment noise counter (used by heartbeat function)
221  noiseCount++;
222 
223  if(currentBit==LONG_GAP)
224  {
225  debugPrint(F("\nGap"));
226  resetMessage();
227  return false;
228  }
229  else
230  {
231  debugPrint(String(currentBit));
232  }
233 
234 
235  //process 'ones', 'zeros' should have been done by resetMessage() function
236  if(currentBit==ONE_BIT) // so process it but only if bit==1
237  {
238  if(bitNumber<=3) //preamble
239  {
240  bitSet(preamble,3-bitNumber);
241  }
242  else if(bitNumber>3 and bitNumber<=11) //ID
243  {
244  //check preamble
245  if(bitNumber==4 and preamble!=5)
246  {
247  debugPrint(F("\nBadPre")); //bad preamble so exit
248  resetMessage();
249  return false;
250  }
251 
252  bitSet(id,11-bitNumber); //set ID
253  }
254  else if(bitNumber==12) //battery
255  {
256  batteryOK=true;
257  }
258  else if(bitNumber==13) //manual transmit
259  {
260  ManualTransmit=true;
261  }
262  else if(bitNumber>13 and bitNumber<=15) //channel
263  {
264  bitSet(channel,15-bitNumber);
265  }
266  else if(bitNumber==16) //negative temperature flag
267  {
268  negativeTemperature=true;
269  }
270  else if(bitNumber>16 and bitNumber<=27) //temperature10 number
271  {
272  bitSet(temperature10,27-bitNumber);
273  }
274  else if(bitNumber>27 and bitNumber<=35) //relative humidity number
275  {
276  bitSet(relativeHumidity,35-bitNumber);
277  }
278  }
279 
280  if(bitNumber<35)
281  {
282  RF_reciever::bitNumber++;
283  }
284  else
285  {
286  //end of message so timestamp and validate
287 
288  timeMicros=micros(); //approximate time of this message
289  RF_reciever::bitNumber=0;
290 
291  debugPrint(F("\nNew Message!\n"));
292 
293 
294  if(validateMessage())
295  {
296  //have a new valid message!
297  return true;
298 
299  }
300  else
301  {
302  //message was invalid, clear message and begin again
303  resetMessage();
304  return false;
305  }
306 
307  }
308 
309  return false; //no complete and valid message yet
310

338 //Classifies the short, long and very long pulses into bits
339 //and then processes any bits found
340 //Returns true if a valid message was found
341 bool RF_reciever::convertPulses2Bits()
342 {
343  unsigned long timeNow; //approximate time interrupt fired
344  unsigned long signal_duration; //microseconds since the last interrupt transition
345  byte zero_or_one;
346  int pinState;
347  bool previousSignalExists; //if 'chirp (i.e. short transmit pulse) occurred before the current silence
348 
349  //Get the time and pinStates stored in the queues
350  while(!queueTime.isEmpty() and !queuePinState.isEmpty() )
351  {
352  timeNow=queueTime.dequeue();
353  pinState=queuePinState.dequeue();
354 
355  if(pinState==HIGH)
356  {
357  //Rising edge logic
358  signal_duration=timeNow-RF_reciever::lastTransitionTimeLow; //how long silence was
359 
360  if(signal_duration<1500)
361  {
362  //too short (i.e. noise), so ignore
363  continue; //go to next pulse in queue
364  }
365  else if(signal_duration<=2500)
366  {
367  //short silence
368  zero_or_one=ZERO_BIT; //a "zero" was received
369  }
370  else if(signal_duration<5000)
371  {
372  //long silence
373  zero_or_one=ONE_BIT; //a "one" was received
374  }
375  else
376  {
377  //very long silence
378  zero_or_one=LONG_GAP; //a long gap at the start or end of a signal
379  }
380 
381 
382  RF_reciever::lastTransitionTimeHigh=timeNow;
383 
384  //if this silence was proceeded by a pulse of around 500uS long then the the bit is valid
385  if(previousSignalExists)
386  {
387  //we have a valid bit
388 
389  previousSignalExists=false; //reset flag
390 
391  //send this bit to become part of a Message
392  bool haveMessage=convertBitToMessage(zero_or_one);
393  if(haveMessage){
394  return true; //have a valid and complete message so call home
395  }
396  }
397 
398  }
399  else
400  {
401  //Falling edge logic (check for around 500uS chirp)
402 
403  signal_duration=timeNow-RF_reciever::lastTransitionTimeHigh; //how long the chirp lasted
404  if(signal_duration<350)
405  {
406  //too short since last went high so ignore (i.e. noise)
407  continue; //process next item in queue
408  }
409  else if(signal_duration>650)
410  {
411  //stayed too long in the HIGH state (i.e. not a chirp, just noise on this frequency?)
412  RF_reciever::lastTransitionTimeLow=timeNow;
413  }
414  else
415  {
416  //else there has been a valid chirp!
417  previousSignalExists=true; //set flag to wait for a silent period now
418 
419  RF_reciever::lastTransitionTimeLow=timeNow;
420  }
421 
422  }
423  }
424  //finished processing all items in the queue and no valid messages so return false
425  return false;
426 
427 }
