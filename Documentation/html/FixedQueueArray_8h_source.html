171 // resize the size of the queue.
172 template<typename T>
173 void FixedQueueArray<T>::resize (const int s) {
174  // defensive issue.
175  if (s <= 0)
176  exit (F("QUEUE: error due to undesirable size for queue size."));
177 
178  // allocate enough memory for the temporary array.
179  T * temp = (T *) malloc (sizeof (T) * s);
180 
181  // if there is a memory allocation error.
182  if (temp == NULL)
183  exit (F("QUEUE: insufficient memory to initialize temporary queue."));
184 
185  // copy the items from the old queue to the new one.
186  for (int i = 0; i < items; i++)
187  temp[i] = contents[(head + i) % size];
188 
189  // deallocate the old array of the queue.
190  free (contents);
191 
192  // copy the pointer of the new queue.
193  contents = temp;
194 
195  // set the head and tail of the new queue.
196  head = 0; tail = items;
197 
198  // set the new size of the queue.
199  size = s;
200 }
